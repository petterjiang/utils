Example
option.textInterval函数的参数例子函数的text参数
    text = {
        list, 当前文字字符串
        listIndex:, 当前字符串在字符串数组中的索引
        index:,当前字符在字符串的索引
        node:,当前字符
        nextIndex 下一个字符在字符串中的索引
    }
    function textInterval(text){
        var time = 3000;
        if(text.node==="。"){
            time = 300
        } else{
            time = 20
        }
        if(text.nextIndex === 0){
            time = 3000
        }
        return time
    }
